#based on odegay commands at https://github.com/odegay/ShoebillPlugin/blob/master/buildmachine/Dockerfile
#used a 32bit version of Ubuntu 16 with a virtual machine in VMware Workstation Player
#it is important that you are logged in with root

#installing the dependencies
apt upgrade
apt update
apt-get install -y openjdk-8-jdk
export JAVA_HOME=/usr/lib/jvm/java-8-openjdk-i386 #if yours is different, modify it here
export PATH=$PATH:$JAVA_HOME/bin
apt-get install -y build-essential cmake wget git
wget https://ftp.gnu.org/pub/gnu/libiconv/libiconv-1.15.tar.gz
tar zxf libiconv-1.15.tar.gz
cd libiconv-1.15
./configure --prefix=/usr/local
make
make installl
ldconfig
apt update

#building the project
wget https://github.com/eupedroosouza/ShoebillPlugin.git
cd ShoebillPlugin
git pull
./build.sh
